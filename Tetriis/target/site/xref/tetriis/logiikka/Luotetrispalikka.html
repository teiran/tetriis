12  public class Luotetrispalikka {
13  
14      private int[][] palikka;
15      private int x;
16      private int y;
17  
18      public Luotetrispalikka(int x, int y) {
19          this.x = x;
20          this.y = y;
21          palikka = luopalikka();
22      }
23  
24      public int[][] getPalikka() {
25          return palikka;
26      }
27  
28      public int[][] luopalikka() {
29          int t = randomWithRange(0, 6); //keksin jostain myöhemmin paremman random muutujan
30          int[][] k = new int[x][y];
31          for (int x1 = 0; x1 < x; x1++) {
32              for (int y1 = 0; y1 < y; y1++) {
33                  k[x1][y1] = 0;
34              }
35          }
36          luoPalikka2(t, k);
37  
38          return k;
39      }
40  
41      private int randomWithRange(int min, int max) {
42          int range = (max - min) + 1;
43          return (int) (Math.random() * range) + min;
44      }
45  
46      private void luoPalikka2(int x, int[][] k) {
47          switch (x) {
48              case 0:  //neliö 
49                  k[0][0] = 1;
50                  k[0][1] = 1;
51                  k[1][0] = 1;
52                  k[1][1] = 1;
53                  break;
54              case 1: //salama oikealle
55                  k[0][0] = 1;
56                  k[0][1] = 2;
57                  k[1][1] = 1;
58                  k[1][2] = 1;
59                  break;
60              case 2:  // suora
61                  for (int y = 0; y < 4; y++) {
62                      k[0][y] = 1;
63                  }
64                  k[0][1] = 2;
65                  break;
66              case 3:  // l:llä oikealle
67                  for (int y = 0; y < 3; y++) {
68                      k[0][y] = 1;
69                  }
70                  k[1][2] = 1;
71                  k[0][1] = 2;
72                  break;
73              case 4:  // t palikka 
74                  for (int i = 0; i < 3; i++) {
75                      k[0][i] = 1;
76                  }
77                  k[1][1] = 1;
78                  k[0][1] = 2;
79                  break;
80              case 5:   //l:llä vasemmalle
81                  for (int i = 0; i < 3; i++) {
82                      k[1][i] = 1;
83                  }
84                  k[0][2] = 1;
85                  k[1][1] = 2;
86                  break;
87              case 6:
88                  k[0][1] = 1;
89                  k[1][0] = 1;
90                  k[1][1] = 2;
91                  k[0][2] = 1;
92                  break;
93          }
94  
95      }
96  }
